```python
from itertools import chain
from loguru import logger
from telegram.ext import Application, ApplicationBuilder,ContextTypes,CommandHandler
from telegram import Update
from ults.read_config import read_bot_config

HASH ,TOKEN,PROXY = read_bot_config()

# 安装 pip install "python-telegram-bot[job-queue]"

#初始化start命令
async def start(update: Update,context : ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text("Hi! Use /set <seconds> to set a timer")

#初始化job成功的输出
async def success_out(context:ContextTypes.DEFAULT_TYPE):
    job = context.job
    await context.bot.send_message(job.chat_id, text=f"Beep! {job.data} seconds are over!")

#清空当前job队列
async def rm_job_queue(name:str,context:ContextTypes.DEFAULT_TYPE):
    current_jop = context.job_queue.get_jobs_by_name(name=name)
    #检测当前 name 队列是否有job
    if not current_jop: 
        return False
    for job in current_jop:
        #队列删除内容
        job.schedule_removal()
    return True

#构建阅读定时器命令
async def set_time(update:Update,context:ContextTypes.DEFAULT_TYPE):
    chat_id = update.effective_message.chat_id
    try:
        first_arg = float(context.args[0])
        logger.info(f"接受到第一个参数{first_arg}")
        if first_arg < 0:
            await update.message.reply_text("Error input must be int and >0!!!!!")
            return
        job_remove = await rm_job_queue(str(chat_id),context=context)
        #设置一个工作队列 并且运行当前工作 第二个参数是时间参数 启动定时器
        context.job_queue.run_once(success_out, first_arg, chat_id=chat_id, name=str(chat_id), data=first_arg)
        #时间执行完毕后 执行回调函数 success_out data 可以通过回调中的 Job.data 访问
        text = "Timer successfully set!"
        if job_remove:
            text += " Old one was removed."
        await update.effective_message.reply_text(text)

    except (IndexError, ValueError):
        await update.effective_message.reply_text("Usage: /set <seconds>")


def init_bot():
    try:
        app = (Application
            .builder()
            .token(TOKEN)
            .proxy(PROXY)
            .get_updates_proxy(PROXY)
            .build())
    except:
        logger.error("检查PROXY设置")
        return
    logger.success("APP构建成功")
    start_hander = CommandHandler(["start","help"],start)
    set_hander = CommandHandler('set',set_time)
    app.add_handler(start_hander)
    app.add_handler(set_hander)
    app.run_polling()
```
