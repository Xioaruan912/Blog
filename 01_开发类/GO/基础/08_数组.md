# 数组

同一类型 Golang中少用数组 多用切片

```go
package main

import "fmt"

func main() {
	var intarray [5]int = [5]int{1, 2, 3, 4, 5}
	fmt.Println(intarray[0])
}

```

# 数组相等与否

```go
package main

import "fmt"

func main() {
	var intarray [5]int = [5]int{1, 2, 3, 4, 5}
	intarray2 := [5]int{1, 2, 3, 4, 6}
	if intarray2 == intarray {
		fmt.Println("e")
	} else {
		fmt.Println("!eq")
	}
}

```
