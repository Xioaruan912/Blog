假设序列为 HSSHHHS 通过栈 变化为SSSHHHH

```C++
#include <stdio.h>

#define MAXSIZE 10
typedef char Elemtype;
typedef struct Stack{
    Elemtype data[MAXSIZE];
    int top;
}Stack;

//基本操作
Stack init_stack(){
    Stack S;
    S.top = -1;
    return S;
}
Elemtype pop(Stack *s){
    Elemtype X = s->data[s->top--];
    return X;
}
void push(Stack *s,Elemtype X){
    s->data[++s->top] = X;
}

//本题操作
void Deal(Stack *S,char String[],char OK[]){
    int j = 0;
    for (int i = 0; String[i] != '\0'; i++){
        if(String[i] == 'H'){
            push(S,String[i]);
        }else if (String[i] == 'S')
        {
            OK[j] = 'S';
            j ++;
        }
    }
    while(S->top != -1){
        OK[j] = pop(S);
        j++;
    }
    OK[j] = '\0';  
}


int main(){
    char OK[100];
    Stack S = init_stack();
    char String[] = "HSSHHHS";
    Deal(&S,String, OK);
    printf("Result: %s\n", OK);
    return 0;
}
```
